#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
const int N=50001;
int C[N],n;
inline int Lowbit(int x){
	return ( x& -x );
}
int Sum(int i){
	int res=0;
	while(i>0){
		res+=C[i];
		i-=Lowbit(i);
	}
	return res;
}
void Modify(int i,int d){
	while(i<N){
		C[i]+=d;
		i+=Lowbit(i);
	}
}
int main()
{
	freopen("in.txt","r",stdin);
	int T,cases=1,i,p,q,x;
	char op[10];
	scanf("%d",&T);
	while(T--){
		printf("Case %d:\n",cases++);
		scanf("%d",&n);
		memset(C,0,sizeof(C));
		for(i=1;i<=n;++i){
			scanf("%d",&x);
			Modify(i,x);
		}
		while(scanf("%s",op) && op[0]!='E'){
		
			scanf("%d%d",&p,&q);
			if(op[0]=='Q'){
				printf("%d\n",Sum(q)-Sum(p-1));
			}
			else if(op[0]=='A'){
				Modify(p,q);
			}
			else
				Modify(p,-q);

		}
	}
	return 0;
}
