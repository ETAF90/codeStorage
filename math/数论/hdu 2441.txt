import java.util.*;
import java.io.*;
import java.math.*;

public class Main {

    public static void main(String[] args) {
        int cnt, now;
        long i;
        BigInteger x;
        Scanner cin = new Scanner(System.in);
        while (cin.hasNextBigInteger()) {
            x = cin.nextBigInteger();
            if (x.compareTo(BigInteger.ZERO) == 0) {
                break;
            }
            cnt = 0;

            for (i = 2; x.compareTo(BigInteger.ONE) != 0; ++i) {
                if (x.mod(BigInteger.valueOf(i)).compareTo(BigInteger.ZERO) == 0) {//ÈôÐ´³Éx.mod(BigInteger.valueOf(i))==BigInteger.ZEROÔò³¬Ê±¡£
                    ++cnt;
                    now = 0;
                    while (x.mod(BigInteger.valueOf(i)).compareTo(BigInteger.ZERO) == 0) {
                        ++now;
                        if (now > 1) {
                            cnt = -1;
                            break;
                        }
                        x = x.divide(BigInteger.valueOf(i));
                    }
                    if (now > 1) {
                        break;
                    }
                }
            }
            // System.out.println("cnt="+cnt);
            if (cnt > 0) {
                System.out.println("yes");
            } else {
                System.out.println("no");
            }
        }
    }
}
