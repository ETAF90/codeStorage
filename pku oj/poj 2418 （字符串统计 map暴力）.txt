int main()
{
	map<string,int> M;
	map<string,int>::iterator it;
	char str[100];
	int sum=0;
	while(gets(str))
	{
		sum++;
		M[str]++;
	}
	for(it=M.begin();it!=M.end();++it)
	{
		cout<<it->first;
		printf(" %.4lf\n",100.0*double(it->second)/double(sum));
	}
	return 0;
}
