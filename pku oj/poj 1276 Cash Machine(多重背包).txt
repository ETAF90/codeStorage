#define N 100001
int f[N],w[N],num[N],n,m;			//¶àÖØ±³°ü
void OneZeroPack(int w,int c)
{
	for(int j=m;j>=c;j--)
		if(f[j]<f[j-c]+w)
			f[j]=f[j-c]+w;
}
void CompletePack(int w,int c)
{
	for(int j=c;j<=m;j++)
		if(f[j]<f[j-c]+w)
			f[j]=f[j-c]+w;
}
void MultiplePack(int w,int c,int number)
{
	if(c*number>=m)
	{
		CompletePack(w,c);
		return ;
	}
	int k=1;
	while(k<number)
	{
		OneZeroPack(k*w,k*c);
		number-=k;
		k<<=1;
	}
	OneZeroPack(number*w,number*c);
}
