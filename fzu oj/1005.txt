#include<stdio.h>
#include<cmath>
#include<string.h>
#define N 202
int main()
{
	int f[N][N],mid[N][N],a[N],cases=1,i,j,k,m,n,tmp;
	while(scanf("%d%d",&n,&m)!=EOF)
	{
		if(m==0 && n==0)
			break;
		for(i=1;i<=n;i++)
			scanf("%d",a+i);
		for(i=1;i<=n;i++)
			for(j=i;j<=n;j++)
			{
				tmp=a[(i+j)/2];
				mid[i][j]=0;
				for(k=i;k<=j;k++)
					mid[i][j]+=abs(a[k]-tmp);
			}
		for(i=1;i<=n;i++)
		{
			f[1][i]=mid[1][i];
			f[i][i]=0;
		}
		for(i=2;i<=m;i++)
			for(j=i;j<=n;j++)
			{
				f[i][j]=mid[1][j];
				for(k=i-1;k<j;k++)//k=i-1£¨ÖØµã£©
					if(f[i][j]>f[i-1][k]+mid[k+1][j])
						f[i][j]=f[i-1][k]+mid[k+1][j];
			}
		printf("Chain %d\nTotal distance sum = %d\n\n",cases++,f[m][n]);
	}
}
