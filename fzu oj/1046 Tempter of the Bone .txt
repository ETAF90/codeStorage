#include<iostream>				//ษ๎หั+ผ๔ึฆ
#include<cmath>
using namespace std;
struct point
{
	int x,y;
};
struct point begin,end;
int dx[]={0,1,0,-1};
int dy[]={1,0,-1,0};
bool flag;
int a[10][10];
void dfs(int t,int x,int y)
{
	if(t==0)
	{
		if(x==end.x && y==end.y)
			flag=true;
		return;
	}
	if(abs(x-end.x)+abs(y-end.y)>t)
		return;
	if(x==end.x && y==end.y)
		return;
	int i,x1,y1;
	for(i=0;i<4;i++)
	{
		x1=x+dx[i];
		y1=y+dy[i];
		if(a[x1][y1]==0)
		{
			a[x1][y1]=1;
			dfs(t-1,x1,y1);
			a[x1][y1]=0;
			if(flag)
				return;
		}
	}

}
int main()
{
	int time,m,n,i,j;
	char x;
	while(scanf("%d%d%d",&m,&n,&time)!=EOF)
	{
		if(time==0 &&m==0 && n==0)
			break;
		for(i=1;i<=m;i++)
		{
			getchar();
			for(j=1;j<=n;j++)
			{
				x=getchar();
				if(x=='S')
				{	begin.x=i,begin.y=j;}
				if(x=='D')
				{	end.x=i;end.y=j;}
				if(x=='X')
					a[i][j]=1;
				else
					a[i][j]=0;
			}
		}

		for(i=0;i<=m+1;i++)
			a[i][0]=a[i][n+1]=1;
		for(j=0;j<=n+1;j++)
			a[0][j]=a[m+1][j]=1;
	/*	for(i=0;i<=m+1;i++)
		{
			for(j=0;j<=n+1;j++)
				printf("%d",a[i][j]);
			printf("\n");
		}
		*/
		if(time>n*m)
		{	printf("NO\n");continue;}
		flag=false;
		dfs(time,begin.x,begin.y);
		if(flag)
			printf("YES\n");
		else
			printf("NO\n");

	}
	return 0;
}
