#include<iostream>
#define N 10000000
using namespace std;
int f[10001],p[501],w[501];
int main()
{
//	freopen("1.txt","r",stdin);
	int T,n,i,j,v0,v;
	scanf("%d",&T);
	while(T--)
	{	
		scanf("%d%d",&v0,&v);
		scanf("%d",&n);
		v=v-v0;
		f[0]=0;
		
		for(i=1;i<=n;i++)
		{
			scanf("%d",p+i);
			scanf("%d",w+i);
		}
		f[0]=0;
		for(i=1;i<=v;i++)
			f[i]=N;
	/*	for(i=0;i<=v;i++)
			cout<<f[i]<<" ";		//打表理解
		cout<<endl;
	*/
		for(i=1;i<=n;i++)
		{
			for(j=w[i];j<=v;j++)
			{
				if(f[j]>f[j-w[i]]+p[i])
					f[j]=f[j-w[i]]+p[i];
			}
		/*	for(j=0;j<=v;j++)
				cout<<f[j]<<" ";
			cout<<endl;
		*/
		}
		if(f[v]==N)
			printf("This is impossible.\n");
		else
			printf("The minimum amount of money in the piggy-bank is %d.\n",f[v]);
		
	}
//	fclose(stdin);
	return 0;
}
