#include<stdio.h>					//¼òµ¥ÉîËÑ
int dx[]={0,1,0,-1,1,-1,-1,1},
	dy[]={1,0,-1,0,1,-1,1,-1},
	ans;
char a[105][105];
void dfs(int x,int y)
{
	int x1,y1;
	for(int i=0;i<8;i++)
	{
		x1=x+dx[i];
		y1=y+dy[i];
		if(a[x1][y1]=='@')
		{
			a[x1][y1]='*';
			dfs(x1,y1);
		}

	}
	

}
int main()
{
	int i,j,m,n;
	while(scanf("%d%d",&m,&n) && m)
	{
		getchar();
		for(i=1;i<=m;i++)
		{
			for(j=1;j<=n;j++)
				scanf("%c",&a[i][j]);
			getchar();
		}
		for(i=0;i<=m+1;i++)
			a[i][0]=a[i][n+1]='*';
		for(j=0;j<=n+1;j++)
			a[0][j]=a[m+1][j]='*';
		/*for(i=0;i<=m+1;i++)
		{
			for(j=0;j<=n+1;j++)
				printf("%c",a[i][j]);
			printf("\n");
		}*/
		ans=0;
		for(i=1;i<=m;i++)
			for(j=1;j<=n;j++)
			{
				if(a[i][j]=='@')
				{
					a[i][j]='*';
					dfs(i,j);
					ans++;
				}
				
			}
		printf("%d\n",ans);
		
	}
	return 0;
}
