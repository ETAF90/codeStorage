#include<iostream>
using namespace std;			//hdoj 1506,1505µÄ¼ÓÇ¿°æ
#define N 1004
#define M 1004
char a[M][N];
int f[M][N],l[N],r[N],m,n;
int fun(int * p)
{
	int max,tmp,i;
	for(i=0;i<=n+1;i++)
	{
		l[i]=i;
		r[i]=i;
	}
	for(i=1;i<=n;i++)				//dp
			while(p[l[i]-1]>=p[i])
				l[i]=l[l[i]-1];
		for(i=n;i>=1;i--)
			while(p[r[i]+1]>=p[i])
				r[i]=r[r[i]+1];
	max=0;
	for(i=1;i<=n;i++)
	{
		tmp=p[i]*(r[i]-l[i]+1);
		if(max<tmp)
			max=tmp;
	}
	return max;
}
int main()
{
	int i,j,k,max,tmp;
	char str[]={'a','b','c'};
	char strno[][3]={'x','b','c','y','a','c','w','a','b'};

	int ans[4];
	while(scanf("%d%d",&m,&n)!=EOF)
	{
		getchar();

		for(i=1;i<=m;i++)
		{
			for(j=1;j<=n;j++)
			{
				scanf("%c",&a[i][j]);
			}
			getchar();
		}
		max=0;
		for(k=0;k<3;k++)
		{	for(i=0;i<=m+1;i++)
		{
			a[i][0]=a[i][n+1]=strno[k][1];
			f[i][0]=f[i][n+1]=-1;
		}
		for(j=0;j<=n+1;j++)
			a[0][i]=a[m+1][i]=strno[k][1];
		for(i=m;i>=1;i--)
			for(j=1;j<=n;j++)
			{
				if(a[i][j]==strno[k][0] || a[i][j]==strno[k][1]|| a[i][j]==strno[k][2])
					f[i][j]=0;
				else
					if(a[i+1][j]==strno[k][0] ||a[i+1][j]==strno[k][1]|| a[i+1][j]==strno[k][2] )
						f[i][j]=1;
					else
						f[i][j]=f[i+1][j]+1;

			}
	/*	
		for(i=0;i<=m+1;i++)
		{
			for(j=0;j<=n+1;j++)
				printf("%d ",f[i][j]);
			printf("\n");
		}
	*/
		ans[k]=0;
		for(i=1;i<=m;i++)
		{
			tmp=fun(f[i]);
			if(ans[k]<tmp)
				ans[k]=tmp;
		}
		if(max<ans[k])
			max=ans[k];
		}
		printf("%d\n",max);
	}
	return 0;
}
