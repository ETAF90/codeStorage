#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
const int N=110;
int g[N][N];
int n,m;
void floyd()
{
	for(int k=1;k<=n;++k)
		for(int i=1;i<=n;++i)
			for(int j=1;j<=n;++j)
				if(g[i][k] && g[k][j])
					g[i][j]=1;
}
void output()
{
	for(int i=1;i<=n;++i)
	{
		for(int j=1;j<=n;++j)
				printf("%d ",g[i][j]);
		printf("\n");
	}
}
int main()
{
	//freopen("in.txt","r",stdin);
	int i,j,x,y,ans;
	bool flag;
	while(scanf("%d%d",&n,&m)!=EOF)
	{
		memset(g,0,sizeof(g));
		while(m--)
		{
			scanf("%d%d",&x,&y);
			g[y][x]=1;
		}
		floyd();
		//output();
		ans=0;
		for(i=1;i<=n;++i)
		{
			flag=1;
			for(j=1;j<=n;++j)
			{
				if(i==j)continue;
				if(!g[i][j] && !g[j][i])
				{
					flag=0;
					break;
				}
			}
			if(flag)
				++ans;
		}
		printf("%d\n",ans);
	}
	return 0;
}
