#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
typedef unsigned __int64 uint64;
typedef unsigned int uint32;

int calculate(int n,int* a)
{
	if(n<=2)
	{
		return a[n];
	}
	else if(n == 3)
	{
		return a[3]+a[1]+a[2];
	}else
	{
		if(a[2]+a[2]<a[n-1]+a[1])
		{
			return calculate(n-2,a)+a[1]+a[2]+a[n]+a[2];
		}else
		{
			return calculate(n-2,a)+a[n]+a[1]+a[n-1]+a[1];
		}
	}

}
void solve()
{
	int n,a[1010];
	scanf("%d",&n);
	for(int i=1; i<=n; ++i)
	{
		scanf("%d",a+i);
	}
	if(n == 1)
	{
		printf("%d\n",a[1]);
		return ;
	}
	sort(a+1,a+1+n);
	printf("%d\n",calculate(n,a));
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		solve();
	}
	return 0;
}
