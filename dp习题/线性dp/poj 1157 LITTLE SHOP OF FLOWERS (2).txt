#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
const int N=101;
const int inf=0x3fffffff;
int g[N][N],dp[2][N];

int main(){

	//freopen("in.txt","r",stdin);
	int i,j,c,n,m;
	scanf("%d%d",&n,&m);
		for(i=1;i<=n;++i){

			for(j=1;j<=m;++j){	
				scanf("%d",&g[i][j]);
			}
		}

		dp[0][0]=0;
		for(i=1;i<=n;++i){
			dp[0][i]=-inf;
		}
		c=0;
		for(i=1;i<=m;++i,c=1-c){

			for(j=1;j<i;++j){

				dp[c][j]=max(dp[1-c][j-1]+g[j][i],dp[1-c][j]);
			}
			dp[c][j]=dp[1-c][j-1]+g[j][i];

         }

		printf("%d\n",dp[1-c][n]);

	return 0;
}
