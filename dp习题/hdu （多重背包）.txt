#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
const int N=110;
int dp[N],c[N],w[N],cnt[N];
int n,v;
void ozPack(int w,int c){
	
	for(int j=v;j>=c;--j){
		dp[j]=max(dp[j],dp[j-c]+w);
	}
}
void comPack(int w,int c){
	
	for(int j=c;j<=v;++j)
		dp[j]=max(dp[j],dp[j-c]+w);

}
void multPack(int w,int c,int num){
	
	if(c*num>=v)
	{
		comPack(w,c);
		return;
	}
	int k=1;
	while(num>k){
		
		ozPack(w*k,c*k);
		num-=k;
		k<<1;
	}
	if(num){
		
		ozPack(num*w,c*num);
	}
}

void solve()
{
	int i,j;
	scanf("%d%d",&v,&n);
	for(i=0;i<n;++i)
	{
		scanf("%d%d%d",c+i,w+i,cnt+i);
	}
	memset(dp,0,sizeof(dp));
	for(i=0;i<n;++i){
		
		multPack(w[i],c[i],cnt[i]);

	}
	printf("%d\n",dp[v]);
}
int main()
{
	int T;
	scanf("%d",&T);
	while(T--){
		
		solve();
	}
	return 0;
}
